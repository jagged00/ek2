pa_apply_lore_height_trait_effect = {
	if = {
		limit = {
			NOT = {
				OR = {
					has_trait = short_1
					has_trait = short_2
					has_trait = short_3
					has_trait = short_4
					has_trait = short_5
					has_trait = average_height
					has_trait = tall_1
					has_trait = tall_2
					has_trait = tall_3
					has_trait = tall_4
					has_trait = tall_5
				}
			}
		}

		# Ensure each character has exactly one race flag before assigning height.
		if = {
			limit = {
				NOT = {
					OR = {
					has_character_flag = race_giant
					has_character_flag = race_atmoran
					has_character_flag = race_orc
					has_character_flag = race_yokudan
					has_character_flag = race_mer
					has_character_flag = race_bosmer
					has_character_flag = race_khajiit
					has_character_flag = race_goblinken
					has_character_flag = race_lilmothiit
					has_character_flag = race_argonian
					has_character_flag = race_kothri
					has_character_flag = race_imga
					has_character_flag = race_breton
					has_character_flag = race_cyrod
					has_character_flag = race_deathlander
					has_character_flag = race_daedra
				}
				}
			}
			if = {
				limit = { OR = { exists = mother exists = real_father exists = father } }
				pa_inherit_single_race_flag_effect = yes
			}
			else = {
				pa_assign_single_race_flag_from_culture_effect = yes
			}
		}

		# Priority 1: inherit from real parents' existing height traits when available.
		if = {
			limit = {
				OR = {
					mother = {
						OR = {
							has_trait = short_1
							has_trait = short_2
							has_trait = short_3
							has_trait = short_4
							has_trait = short_5
							has_trait = average_height
							has_trait = tall_1
							has_trait = tall_2
							has_trait = tall_3
							has_trait = tall_4
							has_trait = tall_5
						}
					}
					real_father = {
						OR = {
							has_trait = short_1
							has_trait = short_2
							has_trait = short_3
							has_trait = short_4
							has_trait = short_5
							has_trait = average_height
							has_trait = tall_1
							has_trait = tall_2
							has_trait = tall_3
							has_trait = tall_4
							has_trait = tall_5
						}
					}
					father = {
						OR = {
							has_trait = short_1
							has_trait = short_2
							has_trait = short_3
							has_trait = short_4
							has_trait = short_5
							has_trait = average_height
							has_trait = tall_1
							has_trait = tall_2
							has_trait = tall_3
							has_trait = tall_4
							has_trait = tall_5
						}
					}
				}
			}
			pa_assign_height_from_parents_traits_effect = yes
		}
		# Priority 2: no parent height traits, use real parents' race flags.
		else_if = {
			limit = {
				OR = {
					exists = mother
					exists = real_father
					exists = father
				}
			}
			pa_assign_height_from_parents_race_effect = yes
		}
		# Priority 3: no parents in scope, use character's own race flag (or derive one from culture as fallback).
		else = {
			pa_assign_height_from_race_effect = yes
		}
	}
}
